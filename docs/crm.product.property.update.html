crm.product.property.update ↗︎

crm.product.property.update(id, fields)

Обновляет существующее свойство товаров.

Параметры

Параметр Описание
id Идентификатор свойства товаров.
fields Набор полей - массив вида array("обновляемое поле"=>"значение"[, ...]), где "обновляемое поле" может принимать значения из возвращаемых методом crm.product.property.fields.

Пример

var id = prompt("Введите ID");
var propertyName = prompt("Введите новое название");
BX24.callMethod(
	"crm.product.property.update",
	{
		id: id,
		fields:
		{
			"NAME": propertyName
		}
	},
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
		{
			console.dir(result.data());
			if(result.more())
				result.next();
		}
	}
);
