bizproc.robot.delete ↗︎

Scope: bizproc Права на выполнение: для всех

bizproc.provider.delete - метод удаляет зарегистрированного робота.

Параметр Описание
CODE Идентификатор робота.

Пример

var params = {
	'CODE': 'robot'
};
BX24.callMethod(
	'bizproc.robot.delete',
	params,
	function(result)
	{
		if(result.error())
			alert('Error: ' + result.error());
		else
			alert("Успешно: " + result.data());
	}
);
