crm.productrow.list ↗︎

Возвращает список товарных позиций по фильтру. Является реализацией списочного метода для товарных позиций. Владелец товарных позиций определяется обязательными полями OWNER_TYPE и OWNER_ID, где OWNER_TYPE - односимвольный код типа ("D" - сделка, "L" - лид), OWNER_ID - идентификатор.

Параметры

Cм. описание списочных методов.

Пример

var ownerType = prompt("Введите тип владельца (D, L)");
var ownerId = prompt("Введите ID владельца");
BX24.callMethod(
	"crm.productrow.list",
	{
		filter:
		{
			"OWNER_TYPE": ownerType,
			"OWNER_ID": ownerId
		}
	},
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
		{
			console.dir(result.data());
			if(result.more())
				result.next();
		}
	}
);
