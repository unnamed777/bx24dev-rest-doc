crm.activity.get ↗︎

crm.activity.get(id)

Возвращает активность по идентификатору.

Параметры

Параметр Описание
id Идентификатор активности.

Пример

var id = prompt("Введите ID");
BX24.callMethod(
	"crm.activity.get",
	{ id: id },
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
			console.dir(result.data());
	}
);
