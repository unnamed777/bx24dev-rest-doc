crm.quote.list ↗︎

Возвращает список предложений по фильтру. Является реализацией списочного метода для предложений.

Параметры

Cм. описание списочных методов.

Пример

BX24.callMethod(
	"crm.quote.list", 
	{ 
		order: { "STATUS_ID": "ASC" },
		filter: { "=COMPANY_ID": 1 },
		select: [ "ID", "TITLE", "STATUS_ID", "OPPORTUNITY", "CURRENCY_ID" ]
	}, 
	function(result) 
	{
		if(result.error())
			console.error(result.error());
		else
		{
			console.dir(result.data());
			if(result.more())
				result.next();
		}
	}
);
