crm.lead.contact.items.set ↗︎

crm.lead.contact.items.set(id, items)

Устанавливает набор контактов, связанных с указанным лидом.

Параметры

Параметр Описание С версии
id Идентификатор лида.
items Набор контактов в виде массива объектов со следующими полями:
  • CONTACT_ID - идентификатор контакта (обязательное поле)
  • SORT - индекс сортировки
  • IS_PRIMARY - флаг первичного контакта

Пример

var id = prompt("Введите ID лида");
BX24.callMethod(
	"crm.lead.contact.items.set",
	{
		id: id,
		items: [{
			"CONTACT_ID": 1010,
			"SORT": 10,
			"IS_PRIMARY": "Y"
		}, {
			"CONTACT_ID": 1020,
			"SORT": 20,
			"IS_PRIMARY": "N"
		}]
	},
	function(result)
	{
		if (result.error())
		{
			console.info("Не удалось добавить контакты к лиду");
			console.error(result.error());
		}
		else
		{
			console.info("Контакты успешно добавлены к лиду");
			console.dir(result.data());
		}
	}
);
