crm.requisite.userfield.list ↗︎

crm.requisite.userfield.list(order, filter)

Возвращает список пользовательских полей реквизита по фильтру. Полное описание полей можно получить вызовом метода crm.userfield.fields

Параметры

Параметр Описание
orderПоля сортировки.
filterПоля фильтра.

Пример

BX24.callMethod(
	"crm.requisite.userfield.list",
	{
		order: { "SORT": "ASC" },
		filter: { "MANDATORY": "N" }
	},
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
		{
			console.dir(result.data());
			if(result.more())
				result.next();
		}
	}
);
