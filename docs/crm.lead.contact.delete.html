crm.lead.contact.delete ↗︎

crm.lead.contact.delete(id, fields)

Удаляет контакт из указанного лида.

Параметры

Параметр Описание С версии
id Идентификатор лида.
fields Объект со следующими полями:
  • CONTACT_ID - идентификатор контакта (обязательное поле)

Пример

var id = prompt("Введите ID лида");
BX24.callMethod(
	"crm.lead.contact.delete",
	{
		id: id,
		fields:
		{
			"CONTACT_ID": 1010
		},
	},
	function(result)
	{
		if (result.error())
		{
			console.info("Не удалось отвязать контакт от лида");
			console.error(result.error());
		}
		else
		{
			console.info("Контакт успешно отвязан от лида");
			console.dir(result.data());
		}
	}
);
