imopenlines.config.update ↗︎

Метод для обновления линии.

Параметры

Метод Описание С версии
CONFIG_ID ID линии (обязательный)
PARAMS Массив параметров Список возможных параметров идентичен списку метода imopenlines.config.add. для обновления (необязательный)

Пример

//imopenlines.config.update
function configUpdate()
{
	var params = {
		CONFIG_ID: 1,
		PARAMS: {
			LINE_NAME: 'New line name',
			...
		}
	};
	BX24.callMethod(
		'imopenlines.config.update',
		params,
		function (result) {
			if (result.error())
				alert("Error: " + result.error());
			else
				alert("Успешно: " + result.data());
		}
	);
}
