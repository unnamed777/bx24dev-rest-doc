crm.lead.contact.add ↗︎

crm.lead.contact.add(id, fields)

Добавляет контакт к указанному лиду.

Параметры

Параметр Описание С версии
id Идентификатор лида.
fields Объект со следующими полями:
  • CONTACT_ID - идентификатор контакта (обязательное поле)
  • SORT - индекс сортировки
  • IS_PRIMARY - флаг первичного контакта

Пример использования

var id = prompt("Введите ID лида");
BX24.callMethod(
	"crm.lead.contact.add",
	{
		id: id,
		fields:
		{
			"CONTACT_ID": 1010,
			"SORT": 10,
			"IS_PRIMARY": "Y"
		}
	},
	function(result)
	{
		if (result.error())
		{
			console.info("Не удалось добавить контакт к лиду");
			console.error(result.error());
		}
		else
		{
			console.info("Контакт успешно добавлен к лиду");
			console.dir(result.data());
		}
	}
);
