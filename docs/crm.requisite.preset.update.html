crm.requisite.preset.update ↗︎

crm.requisite.preset.update(fields)

Обновление шаблона реквизита.

Параметры

Параметр Описание
fieldsНабор полей - массив вида array("поле"=>"значение"[, ...]), содержащий значения полей шаблона. Где "поле" может принимать значения из возвращаемых методом crm.requisite.preset.fields.
Примечание: чтобы узнать требуемый формат полей, выполните метод crm.requisite.preset.fields и посмотрите формат пришедших значений этих полей.

Пример

var id = prompt("Введите ID");
BX24.callMethod(
	"crm.requisite.preset.update",
	{
		id: id,
		fields:
		{
			"ENTITY_TYPE_ID": 8,
			"COUNTRY_ID": 1,
			"NAME": "ИП (архив)",
			"ACTIVE": "N"
		}
	},
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
		{
			console.info(result.data());
		}
	}
);
