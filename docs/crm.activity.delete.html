crm.activity.delete ↗︎

crm.activity.delete(id)

Удаляет активность.

Параметры

Параметр Описание
id Идентификатор активности.

Пример

var id = prompt("Введите ID");
BX24.callMethod(
	"crm.activity.delete",
	{ id: id },
	function(result)
	{
		if(result.error())
			console.error(result.error());
		else
			console.info(result.data());
	}
);
