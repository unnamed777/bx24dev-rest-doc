crm.lead.contact.items.delete ↗︎

crm.lead.contact.items.delete(id)

Очищает набор контактов, связанных с указанным лидом.

Параметры

Параметр Описание С версии
id Идентификатор лида.

Пример

var id = prompt("Введите ID лида");
BX24.callMethod(
	"crm.lead.contact.items.delete",
	{
		id: id
	},
	function(result)
	{
		if (result.error())
		{
			console.info("Не удалось очистить набор контактов для лида");
			console.error(result.error());
		}
		else
		{
			console.info("Все контакты для лида успешно удалены");
			console.dir(result.data());
		}
	}
);
