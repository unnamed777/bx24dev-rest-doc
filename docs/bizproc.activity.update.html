bizproc.activity.update ↗︎

Scope: bizproc Права на выполнение: для всех

Метод позволяет обновить поля действия. Параметры метода аналогичны bizproc.activity.add.

Пример

function updateActivity1()
{
	var params = {
		'CODE': 'hash',
		'FIELDS': {
			'DOCUMENT_TYPE': '',
			'FILTER': ''
		},
	};
	BX24.callMethod(
		'bizproc.activity.update',
		params,
		function(result)
		{
			if(result.error())
				alert("Error: " + result.error());
			else
				alert("Успешно: " + result.data());
		}
	);
}
