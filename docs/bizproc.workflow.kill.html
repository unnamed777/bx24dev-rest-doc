bizproc.workflow.kill ↗︎

Scope: bizproc Права на выполнение: для всех

bizproc.workflow.kill - метод удаляет запущенный бизнес-процесс.

Параметры

Параметр Описание
ID Идентификатор бизнес-процесса.

Пример

function killWf(id, cb)
{
	var params = {ID: id};
	BX24.callMethod(
		'bizproc.workflow.kill',
		params,
		function(result)
		{
			if(result.error())
				alert("Error: " + result.error());
			else if (cb)
				cb();
		}
	);
}
