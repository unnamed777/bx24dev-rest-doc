tasks.api.scrum.kanban.addStage ↗︎

Scope: task Права на выполнение: для всех 

tasks.api.scrum.kanban.addStage(fields)

Метод создаёт стадию канбана Скрама.

Параметры

Параметр Описание
fields Поля, соответствующие доступному списку полей tasks.api.scrum.kanban.getFields.

Поля name и sprintId обязательные.

Поле sort кратно 100.

Доступные значения для поля type: (NEW, WORK, FINISH).

Примеры

const sprintId = 2;
let name = 'firstColumn';
let sort = 100;
let type = 'NEW';
let color = '00C4FB';
BX24.callMethod(
	'tasks.api.scrum.kanban.addStage',
{
	fields: {
		sprintId: sprintId,
		name: name,
		sort: sort,
		type: type,
		color: color,
	}
},
	function(res)
	{
		console.log(res);
	}
);
name = 'secondColumn';
sort = 200;
type = 'FINISH';
color = '75D900';
BX24.callMethod(
	'tasks.api.scrum.kanban.addStage',
{
	sprintId: sprintId,
	fields: {
		name: name,
		sort: sort,
		type: type,
		color: color,
	}
},
	function(res)
	{
		console.log(res);
	}
);
